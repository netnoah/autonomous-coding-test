```jsx
// HP bar gradient calculation
const getHpBarColor = (percent) => {
  if (percent >= 80) {
    return `hsl(120, 70%, 45%)` // Pure green
  } else if (percent >= 40) {
    // Yellow to green gradient (60-120 hue)
    const ratio = (percent - 40) / (80 - 40)
    const hue = 60 + (ratio * 60)
    return `hsl(${hue}, 80%, 45%)`
  } else {
    // Red to yellow gradient (0-60 hue)
    const ratio = Math.max(0, (percent - 0) / (40 - 0))
    const hue = ratio * 60
    return `hsl(${hue}, 80%, 45%)`
  }
}

const hpBarColor = getHpBarColor(hpPercent)
```
